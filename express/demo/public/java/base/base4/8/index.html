# 运动框架

在网页开发中无时无刻不存在运动

他们是怎么实现的呢?

# 有请我们的运动框架 闪亮登场

/**
 * Created by Administrator on 2016/2/28.
 */
/**
 *
 * @param obj  运动的对象
 * @param json {属性}  {top:200 , width:200 , height:500 left}  json
 * @param fn   回调  function(){alert()} 警察
 */
function fnMove(obj, json, fn) {
    // 获取样式 : 兼容
    HTMLElement.prototype.__defineGetter__("currentStyle", function () {
        return this.ownerDocument.defaultView.getComputedStyle(this, null);
    });

    //1:清除
    clearInterval(obj.timer);//清除正在执行的定时器
    //2:速度
    var iSpeed = 0;
    //3:开启定时器
    obj.timer = setInterval(function () {
        //是否运动完成
        var bOver = true; //假设运动完成
        //forin
        for (var attr in json) {//attr:width  opacity
            var iCur = 0;
            if (attr === "opacity") {//透明度   0--100  57
                //round 四舍五入
                iCur = Math.round(parseFloat(getStyle(obj, "opacity")) * 100);
            } else {
                iCur = parseInt(getStyle(obj, attr));
            }

            //4:定义速度  (目标值-当前值)/变化率

            //(500-200)/5
            iSpeed = (json[attr] - iCur) / 5;//获取相对速度
            //5:上下取整
            iSpeed = iSpeed > 0 ? Math.ceil(iSpeed) : Math.floor(iSpeed);//上下取整
            //6:是否到达目标点
            if (iCur !== json[attr]) {
                //没有运动结束
                bOver = false; //如果某一个属性没有完成运动，关闭按钮
                //
                if (attr === "opacity") {
                    //IE 赋值
                    obj.style.filter = 'alpha(opacity:' + iCur + iSpeed + ')';
                    //别的浏览器
                    obj.style.opacity = (iCur + iSpeed) / 100;
                } else {
                    //其他属性
                    obj.style[attr] = iCur + iSpeed + "px";
                }
            }
        }
        //console.log(bOver)
        if (bOver) {//运动完成
            clearInterval(obj.timer);
            //fn:null 如果true  会触发fn()回调;

            // && : 前面成立 后面才会执行
            //un... false
            //如果传入了fn 就会执行fn();
            fn && fn();//如果有回调方法，就调用

            //回调:
            //警察  小偷
            //观察者 被观察者   观察者模式 回调
        }
    }, 30)

}

function getId(str) {
    return document.getElementById(str);
}

function getStyle(obj, attr) {//获取css样式
    return obj.currentStyle[attr];

}

        fnMove(oOne, {width: 400},function () {

                fnMove(oOne, {width: 800})
            })
        })

这里面我们传入的3个参数

# 实际运用案例二 横向轮播

<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>Title</title>
    <style type="text/css">
        * {
            margin: 0;
            padding: 0;
        }

        #box {
            position: relative;
            width: 960px;
            height: 450px;
            /*
              内容: 行内 /行内块才能居中
              text-alien:center
            */

            /*居中本身:块级元素*/
            margin: 0 auto;
            /*内容溢出隐藏*/
            overflow: hidden;

        }

        /*去除小点点*/
        li {
            list-style: none;
        }

        .list {
            /*left  right 控制位置*/
            position: absolute;
            /*横向排列*/
            width: 4800px;
        }

        .list > li {
            width: 960px;
            height: 450px;
            display: inline-block;
            float: left;
        }

        .btn {
            position: absolute;
            bottom: 10px;
            /*小点能不能居中*/
            text-align: center;

            /*拉开*/
            left: 0;
            right: 0;
        }

        .btn > li {
            width: 18px;
            height: 16px;
            background-image: url("img/dot.png");
            background-position: 0 16px;

            /*展示在下面*/

            /*横向排列*/
            display: inline-block;
        }

        .change_btn {
            position: absolute;
            width: 60px;
            height: 90px;
            background-image: url("img/btn.jpg");
            top: 40%;
        }

        #btn_left {
            left: 0;
            background-position: 0 180px;
        }

        #btn_right {
            right: 0;
            background-position: 0 90px;
        }

        /*display: none:隐藏*/
        #box > p {
            display: none;

        }

        /*鼠标移上出现 移走不出现:*/
        #box:hover > p {
            display: block;
        }
    </style>
</head>
<body>
<!--1:布局-->

<div id="box">
    <ul class="list">
        <li><img src="img/1.jpg" alt=""></li>
        <li><img src="img/2.jpg" alt=""></li>
        <li><img src="img/3.jpg" alt=""></li>
        <li><img src="img/4.jpg" alt=""></li>
        <li><img src="img/5.jpg" alt=""></li>
    </ul>
    <ul class="btn">
        <li></li>
        <li></li>
        <li></li>
        <li></li>
        <li></li>
    </ul>

    <p id="btn_left" class="change_btn"></p>
    <p id="btn_right" class="change_btn"></p>
</div>

</body>

<script src="js/move.js"></script>
<script>


    let oBtnLeft = document.querySelector("#btn_left");
    let oBtnRight = document.querySelector("#btn_right");
    let oUlList = document.querySelector(".list");
    let oLiList = oUlList.getElementsByTagName("li");


    let oUlBtn = document.querySelector(".btn");
    let oLiBtn = oUlBtn.getElementsByTagName("li");


    let length = oLiList.length;

    //开始的位置:0
    var iNow = 0;

    oBtnLeft.onclick = function () {
        //控制 oUlList 的 left
        iNow--;
        if (iNow < 0) {
            iNow = length - 1;
        }
        fnMove(oUlList, {left: -iNow * 960});
        changeDot(iNow);
    };


    oBtnRight.onclick = function () {
        //控制 oUlList 的 left
        iNow++;
        if (iNow > length - 1) {
            iNow = 0;
        }
        fnMove(oUlList, {left: -iNow * 960});
        changeDot(iNow);
    };


    function changeDot(index) {
        //1:下标
        //2:小圆点的背景定位
        //一开始:第一个编程蓝色

        //所有的都变成灰色
        for (let i = 0; i < length; i++) {
            oLiBtn[i].style.backgroundPosition = "0 16px";
        }
        //选中的变成蓝色
        oLiBtn[index].style.backgroundPosition = "0 0";
    }

    changeDot(0);

</script>
</html>
