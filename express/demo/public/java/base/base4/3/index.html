# javascript

# 1:什么是JavaScript

  • JavaScript是一种基于对象和事件驱动的、并具有安全性能的脚本语言

# 2:和java对比

java javascript
语言类型 强类型 弱类型(str="" 字符串)
数据类型 基本,引用 5种(boolean number string...)
控制语句 if,else,switch,for while 基本没区别
函数 修饰符 返回值类型 方法名(参数){方法体 返回值return} function(参数){方法体}
注释 //单行注释 , /**/多行注释 一样

# 2:JavaScript特点

  • 向HTML页面中添加交互行为
  • 脚本语言，语法和Java类似
  • 解释性语言，边执行边解释

# 3:分类

  • ecmascript : 语法
  • DOM : 操作的是文档对象
  • BOM : 浏览器 打开窗口

# 4:网页中引用JavaScript的方式

  • 使用<script>标签
    <script language="javascript"></script>

  • 外部JS文件
    <script src="hello.js" language="javascript"></script>

  • 直接在HTML标签中
onclick="javascript:alert('欢迎你');"

# 7.3 arguments的使用

    //1:定义函数
    function sum3() {
        document.write("和:");
        //累加传过来的参数  arguments.length长度
        // document.write(arguments.length);
        var result3 = 0;

# 8:DOM+事件

DOM: Document Object Model 文档对象模型
重要性: DOM>BOM 难度: DOM>BOM DOM作用:

  • 1:操作CSS
  • 2:操作HTML 别名:节点

# 8.1 获取DOM元素

  • document.getElementById() : 通过id获取元素
  • *document.getElementsByName() : 通过名字 name
  • *document.getElementsByTagName() : 标签
  • document.getElementsByClassName() :通过class 获取元素

共同点: 都需要通过 document调用

<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>Title</title>
    <style>
        .box {

    </style>
</head>

<h1>标题标题标题</h1>

    /*
    * 操作上述的元素
    * 1:获取节点  id

    var oBtn = document.getElementById("btn");

    var oBtn2 = document.getElementById("btn2");

        alert("真的点我啊~~~");

    //拿到的是标签
    var oInput = document.getElementsByName("my_name")[0];

    // 找到class
    var oBox1 = document.getElementsByClassName("box")[0];
    oBtn2.onclick = function () {
        //样式 变成红色
        oBox1.style.backgroundColor = "red";
    }

</script>
</body>
</html>

# 8.2案例:购物车加减

<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>Title</title>
</head>
<body>
<span>单价: </span>
<span class="price_one">40</span><br>
<span>数量:</span>
<br>
<button class="btn1">-</button>
<!--disabled:不能写-->
<input id="num" type="text" disabled="disabled" style="width: 30px;text-align: center" value="10">
<button class="btn2">+</button>

<br>
<span>总价:</span>
<span class="price_sum">0</span>
<span>元</span>


<script>

    /*
    * 1:获取节点
    * */
    var price_one = document.getElementsByClassName("price_one")[0];
    var btn1 = document.getElementsByClassName("btn1")[0];
    var btn2 = document.getElementsByClassName("btn2")[0];
    var num = document.getElementById("num");
    var price_sum = document.getElementsByClassName("price_sum")[0];

    /*
    * 2:操作
    *
    * */
    btn1.onclick = function () {
        //1: 获取num里面的值
        var value = parseInt(num.value);
        console.log(typeof value);

        //2: 转换成数值类型

        //2: 判断num<1 没有效果
        if (value <= 1) {
            return;
        }
        //3: 点击后重新给num赋值
        var new_value = value -= 1;
        num.value = new_value;
    };

    btn2.onclick = function () {
        //1: 获取num里面的值
        var value = parseInt(num.value);
        console.log(typeof value);

        //2: 转换成数值类型

        //3: 点击后重新给num赋值
        var new_value = value += 1;
        num.value = new_value;
    };


</script>

</body>
</html>
