# 一、事件:

称 说明
onload 一个页面或一幅图像完成加载
onlick 鼠标单击某个对象
onmouseover 鼠标移到某元素上
onmousedown 鼠标点击到某元素上
onmouseup 鼠标点击后抬起某元素
onmouseover 鼠标移出某元素
onkeydown 某个键盘按键被按下
onchange 域的内容被改变

# 1.1鼠标移上

<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>Title</title>
    <style type="text/css">
        #box {
            width: 200px;
            height: 200px;
            background-color: #ff600c;
        }
    </style>

    <script>
        /*当页面加载完成之后 才会执行的操作*/
        window.onload = function () {
            let oBox = document.getElementById("box");
            //oBox.style.backgroundColor = "red";

            //鼠标移上事件
            oBox.onmouseover = function () {
                // this:java:类的对象
                // this:javascript:方法的拥有者 -> oBox
                // console.log(this);

                // 慎用
                this.style.backgroundColor = "green";
            };
            oBox.onmouseout = function () {
                // this:java:类的对象
                // this:javascript:方法的拥有者 -> oBox
                // console.log(this);

                // 慎用
                this.style.backgroundColor = "#ff600c";
            };


        }

    </script>
</head>
<body>
<div id="box"></div>

</body>
</html>

# 1.2 鼠标移动

<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>Title</title>
    <style type="text/css">
        #box {
            width: 100px;
            height: 100px;
            background-color: #ff600c;
            position: absolute;
            /*
            top
            left
            right
            bottom
            */
        }
    </style>

    <script>
        /*当页面加载完成之后 才会执行的操作*/
        window.onload = function () {
            let oBox = document.getElementById("box");

            //1:获取鼠标位置 x y
            //event: 事件 :位置
            //over:只执行一次
            //move:移动
            document.onmousemove = function (event) {
                // console.log(event);
                let mouseX = event.clientX;
                let mouseY = event.clientY;
                // console.log("横坐标:",mouseX);
                // console.log("纵坐标:",mouseY);

                // console.log(oBox.clientHeight);
                console.log(mouseY);
                console.log(mouseX);
                oBox.style.top = mouseY - oBox.clientHeight / 2 + "px";
                oBox.style.left = mouseX - oBox.clientWidth / 2 + "px";

            }


        }

    </script>
</head>
<body>
<div id="box"></div>

</body>
</html>

setTimeout( ) 在指定的毫秒数后调用函数或计算表达式
setInterval( ) 按照指定的周期（以毫秒计）来调用函数或表达式

# 3.1 Date函数

function disptime(){
        var today = new Date(); //获得当前时间
        var hh = today.getHours();//获得小时、分钟、秒
        var mm = today.getMinutes();//获得分钟
        var ss = today.getSeconds();//获得秒
        /*设置div的内容为当前时间*/
        document.getElementById("myclock").innerHTML="现在是:"+hh +":"+mm+": "+ss;
}
